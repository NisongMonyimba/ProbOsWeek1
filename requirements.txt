# ── Scientific computing core ─────────────────────────────────────────────────
# numpy : fast multi-dimensional arrays, random number generators, math ops.
#         This is the backbone of everything. We use np.random.default_rng()
#         (the modern PCG64 generator) — NEVER the legacy np.random.seed().
numpy>=1.26

# scipy : scientific algorithms — statistics, integration, special functions.
#         Used for: erfinv() (inverse error function in Normal.ppf),
#         gaussian_kde (kernel density estimate in Empirical distribution),
#         beta distribution functions.
scipy>=1.12

# matplotlib : plotting library. Used for histograms and example visualisations.
matplotlib>=3.8

# ── Simulation tools ─────────────────────────────────────────────────────────
# SALib : Saltelli sensitivity analysis library.
#         Used in Week 4 for Sobol' indices.
#         Listed here now so the install step does not change later.
SALib>=1.5

# numba : JIT compiler for Python. Speeds up the Monte Carlo loop by 10-50x.
#         Used in Week 3. Listed here for consistency.
numba>=0.59

# plotly : interactive visualisation library.
#          Used in Week 3 for trajectory cloud plots.
plotly>=5.18

# ── Development and quality tools ─────────────────────────────────────────────
# pytest     : the Python test runner. Run with: pytest python/tests/ -v
pytest>=7.4

# pytest-cov : measures which lines of code are covered by tests.
#              Run with: pytest --cov=python/src --cov-report=term-missing
pytest-cov>=4.1

# hypothesis : property-based testing. Generates random inputs automatically.
#              Used in Week 3 to test the Monte Carlo engine.
hypothesis>=6.92

# mypy : static type checker. Verifies type annotations are consistent.
#        Run with: mypy python/src/ --strict
#        Our goal: zero mypy errors at all times.
mypy>=1.8

# ruff : extremely fast Python linter and formatter (replaces flake8 + isort).
#        Run with: ruff check python/
#        Our goal: zero ruff warnings at all times.
ruff>=0.3
